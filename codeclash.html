// Binary Tree Maximum Path Sum // Return the maximum path sum through any path. function maxPathSum(root) { let maxSum = -Infinity; const dfs = (node) => { if (!node) return 0; const left = Math.max(dfs(node.left), 0); const right = Math.max(dfs(node.right), 0); maxSum = Math.max(maxSum, node.val + left + right); return node.val + Math.max(left, right); }; dfs(root); return maxSum; }
